### vue-news 프로젝트 생성
~~~
// 프로젝트 생성
vue create vue-news

// 라우터 설치
yarn add vue-router

// scss 사용
yarn add sass-loader 
yarn add node-sass

// axios
yarn add axios

// vuex : 상태 관리 도구로 여러 컴포넌트에서 굥유되는 데이터속성
yarn add vuex
~~~
